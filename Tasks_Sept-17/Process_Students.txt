import json

with open('students.json', 'r') as file:
    students = json.load(file)


print("All Students:")
for s in students:
    print("-", s["name"])



print("\nMarks Summary:")
for s in students:
    total = sum(s["marks"].values())  # Correct way to sum all mark values
    avg = total / len(s["marks"])
    print(f"{s['name']} â†’ Total: {total}, Average: {avg:.2f}")
